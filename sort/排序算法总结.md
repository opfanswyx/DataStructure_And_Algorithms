```c
void swap(int *a, int *b)
{
    int temp = *a;
    *a = *b;
    *b = temp;
}
```

### 快速排序
```c
void QuickSort(int arr[],int left,int right)
{
    if(left>=right)                     //如果左边索引大于或等于右边索引，则该组序列整理完毕
        return;
    
    int i=left;
    int j=right;
    int key=arr[i];
    while(i<j)
    {
        while((i<j)&&(key<=arr[j]))     //从右向左找小于key的数 不稳定排序
        {
            j--;
        }
        arr[i]=arr[j];                  //将该数赋给arr[i]

        while((i<j)&&(key>=arr[i]))     //从左向右找大于key的数
        {
            i++;
        }
        arr[j]=arr[i];                  //将该数赋给arr[j]
    }
    arr[i]=key;                         //将key赋值给arr[i]
    QuickSort(arr,left,i-1);            //递归调用key两边的新区间，对新区间在调用快排
    QuickSort(arr,i+1,right);
}
```

### 直接插入排序
```c
void insert_sort(int a[], int n)
{
    int i, j, k;

    for (i = 1; i < n; i++)
    {  
        for (j = i - 1; j >= 0; j--)    //为a[i]在前面的a[0...i-1]有序区间中找一个合适的位置
            if (a[j] < a[i])
                break;

        if (j != i - 1)                 //如找到了一个合适的位置
        {   
            int temp = a[i];            //将比a[i]大的数据向后移
            for (k = i - 1; k > j; k--)
                a[k + 1] = a[k];
            a[k + 1] = temp;            //将a[i]放到正确位置上
        }
    }
}
```
